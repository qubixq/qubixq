using namespace std;

int main() {
    string languages[] = {"JavaScript", "HTML", "CSS", "C#", "Bash", "Gentoo Linux", "MySQL", "Python"};
    int size = sizeof(languages) / sizeof(languages[0]);
    cout << "I write ";

        if (i < size - 2) {
            cout << ", ";
        } else if (i == size - 2) {
            cout << " and ";
        }
    }
    cout << ".\n";
    return 0;
}
